# 12.4 AP Subledger Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship the Operating-Expense / AP subledger described in spec `docs/superpowers/specs/2026-05-20-12.4-ap-subledger-design.md` — vendor master + multi-line bills with WT + M:N payment allocations + cheque support + attachments + recurring templates + cron + 14 admin routes, all bridged into the existing GL via two new source kinds.

```sql
  CONSTRAINT bills_gross_nonneg       CHECK (gross_amount >= 0),
  CONSTRAINT bills_paid_nonneg        CHECK (paid_amount >= 0)

-- Bill line items.
CREATE TABLE public.bill_lines (
  id            uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  bill_id       uuid NOT NULL REFERENCES public.bills(id) ON DELETE CASCADE,
  line_no       int NOT NULL,
  description   text,
  amount_php    numeric(12,2) NOT NULL,
  account_id    uuid NOT NULL REFERENCES public.chart_of_accounts(id),
  created_at    timestamptz NOT NULL DEFAULT now(),
  CONSTRAINT bill_lines_line_no_positive CHECK (line_no >= 1),
```

**NOTE on forward-reference:** `bills.template_id` references `recurring_bill_templates`, which is created later in Section 2. PG accepts forward FK references inside a single migration block only if the target table exists *at the time of the constraint check* — for CREATE TABLE with REFERENCES, the target must already exist. If the target table doesn't exist at CREATE TABLE time, the migration fails. Reorder: create `recurring_bill_templates` BEFORE `bills`, OR create `bills` without the `template_id` FK constraint and add the constraint after templates exist via `ALTER TABLE bills ADD CONSTRAINT`.

Going with the ALTER TABLE approach — keeps the conceptual order (bills before templates) while satisfying PG's ordering requirement.

Update the `bills` table create to remove the `REFERENCES` clause inline:

```sql
  template_id              uuid,    -- FK added in Section 2c below
```

```sql
-- Recompute bills.paid_amount + flip status bidirectionally
-- (posted ↔ partially_paid ↔ paid). Only counts non-voided allocations.
CREATE OR REPLACE FUNCTION public.ap_recompute_bill_paid_and_status()
RETURNS TRIGGER
LANGUAGE plpgsql
SECURITY DEFINER
SET search_path = public, pg_temp
AS $$
DECLARE
  v_bill_id        uuid := COALESCE(NEW.bill_id, OLD.bill_id);
  v_paid           numeric(12,2);
  v_net_payable    numeric(12,2);
  v_current_status text;
  v_new_status     text;
BEGIN
  SELECT
    COALESCE(SUM(allocated_amount) FILTER (WHERE voided_at IS NULL), 0),
    b.net_payable,
    b.status
  INTO v_paid, v_net_payable, v_current_status
  FROM public.bills b
  LEFT JOIN public.bill_payment_allocations a ON a.bill_id = b.id
  WHERE b.id = v_bill_id
  GROUP BY b.net_payable, b.status;

  IF v_current_status IN ('draft', 'voided') THEN
    -- Don't touch status; just update paid_amount.
    UPDATE public.bills SET paid_amount = v_paid WHERE id = v_bill_id;
    RETURN NULL;
  END IF;

  -- Determine new status (bidirectional).
  IF v_paid >= v_net_payable AND v_net_payable > 0 THEN
    v_new_status := 'paid';
  ELSIF v_paid > 0 THEN
    v_new_status := 'partially_paid';
  ELSE
    v_new_status := 'posted';
  END IF;

  UPDATE public.bills
  SET paid_amount = v_paid, status = v_new_status
  WHERE id = v_bill_id;

  RETURN NULL;
END;
$$;
```

```sql
-- ==========================================================================
-- Section 4 — Constraint trigger for allocation invariants
-- (deferred to transaction commit).
-- ==========================================================================

CREATE OR REPLACE FUNCTION public.ap_validate_bill_payment_allocations()
RETURNS TRIGGER
LANGUAGE plpgsql
SECURITY DEFINER
SET search_path = public, pg_temp
AS $$
DECLARE
  v_payment_id      uuid := COALESCE(NEW.payment_id, OLD.payment_id);
  v_bill_id         uuid := COALESCE(NEW.bill_id, OLD.bill_id);
  v_alloc_sum_pay   numeric(12,2);
  v_alloc_sum_bill  numeric(12,2);
  v_payment_amount  numeric(12,2);
  v_bill_net        numeric(12,2);
  v_bill_status     text;
  v_payment_vendor  uuid;
  v_bill_vendor     uuid;
BEGIN
  -- Skip if payment row is gone (cascade delete scenario).
  SELECT amount_php, vendor_id INTO v_payment_amount, v_payment_vendor
  FROM public.bill_payments WHERE id = v_payment_id;
  IF v_payment_amount IS NULL THEN
    RETURN NULL;
  END IF;

  -- P0014: SUM(allocated_amount WHERE voided_at IS NULL) per payment = payment.amount_php
  SELECT COALESCE(SUM(allocated_amount), 0) INTO v_alloc_sum_pay
  FROM public.bill_payment_allocations
  WHERE payment_id = v_payment_id AND voided_at IS NULL;

  IF v_alloc_sum_pay <> v_payment_amount THEN
    RAISE EXCEPTION 'Allocation total (%) does not match payment amount (%).',
      v_alloc_sum_pay, v_payment_amount
      USING ERRCODE = 'P0014';
  END IF;

  -- P0015 + P0016 + P0017 only matter if a bill is involved.
  IF v_bill_id IS NOT NULL THEN
    -- P0015: SUM per bill <= bill.net_payable
    SELECT COALESCE(SUM(allocated_amount), 0) INTO v_alloc_sum_bill
    FROM public.bill_payment_allocations
    WHERE bill_id = v_bill_id AND voided_at IS NULL;

    SELECT net_payable, status, vendor_id INTO v_bill_net, v_bill_status, v_bill_vendor
    FROM public.bills WHERE id = v_bill_id;

    IF v_alloc_sum_bill > v_bill_net THEN
      RAISE EXCEPTION 'Allocation total (%) exceeds bill outstanding (%).',
        v_alloc_sum_bill, v_bill_net
        USING ERRCODE = 'P0015';
    END IF;

    -- P0016: vendor match
    IF v_payment_vendor <> v_bill_vendor THEN
      RAISE EXCEPTION 'Allocation bill vendor (%) does not match payment vendor (%).',
        v_bill_vendor, v_payment_vendor
        USING ERRCODE = 'P0016';
    END IF;

    -- P0017: status must be posted/partially_paid/paid (we tolerate paid because
    -- recompute may have flipped status to paid mid-transaction).
    IF v_bill_status NOT IN ('posted', 'partially_paid', 'paid') THEN
      RAISE EXCEPTION 'Cannot allocate to bill in status %.', v_bill_status
        USING ERRCODE = 'P0017';
    END IF;
  END IF;

  RETURN NULL;
END;
$$;
```

```sql
-- ==========================================================================
-- Section 6 — Bridge triggers (post JEs to journal_entries / journal_lines).
-- ==========================================================================

-- Fires when bills.status transitions from 'draft' to 'posted'.
-- Posts a balanced JE: DR each bill_line.account_id, CR 2340 (if WT), CR 2100 (net).
CREATE OR REPLACE FUNCTION public.ap_bill_post_bridge()
RETURNS TRIGGER
LANGUAGE plpgsql
SECURITY DEFINER
SET search_path = public, pg_temp
AS $$
DECLARE
  v_je_id        uuid;
  v_je_number    text;
  v_vendor_name  text;
  v_ap_acct      uuid;
  v_wt_acct      uuid;
  v_line_no      int := 1;
  v_line         RECORD;
BEGIN
  -- Lookup vendor name + the two fixed CoA accounts.
  SELECT name INTO v_vendor_name FROM public.vendors WHERE id = NEW.vendor_id;
  SELECT public.coa_uuid_for_code('2100') INTO v_ap_acct;
  IF NEW.wt_amount > 0 THEN
    SELECT public.coa_uuid_for_code('2340') INTO v_wt_acct;
  END IF;

  -- Create draft JE header.
  INSERT INTO public.journal_entries (
    posting_date, description, status, source_kind, source_id, created_by
  ) VALUES (
    NEW.bill_date,
    format('Bill %s — %s — %s', NEW.bill_number, v_vendor_name, COALESCE(NEW.description, '')),
    'draft',
    'bill_post',
    NEW.id,
    NEW.posted_by
  ) RETURNING id, entry_number INTO v_je_id, v_je_number;

  -- DR lines (one per bill_line, preserving traceability).
  FOR v_line IN
    SELECT line_no, amount_php, account_id, description
    FROM public.bill_lines WHERE bill_id = NEW.id ORDER BY line_no
  LOOP
    INSERT INTO public.journal_lines (
      journal_entry_id, line_order, account_id, debit_amount, credit_amount, description
    ) VALUES (
      v_je_id, v_line_no, v_line.account_id, v_line.amount_php, 0,
      COALESCE(v_line.description, format('Bill line %s', v_line.line_no))
    );
    v_line_no := v_line_no + 1;
  END LOOP;

  -- CR 2340 (WT) if applicable.
  IF NEW.wt_amount > 0 THEN
    INSERT INTO public.journal_lines (
      journal_entry_id, line_order, account_id, debit_amount, credit_amount, description
    ) VALUES (
      v_je_id, v_line_no, v_wt_acct, 0, NEW.wt_amount,
      format('WT %s%% (%s)', (NEW.wt_rate * 100), COALESCE(NEW.wt_classification, ''))
    );
    v_line_no := v_line_no + 1;
  END IF;

  -- CR 2100 (AP — Trade) for net_payable.
  INSERT INTO public.journal_lines (
    journal_entry_id, line_order, account_id, debit_amount, credit_amount, description
  ) VALUES (
    v_je_id, v_line_no, v_ap_acct, 0, NEW.net_payable, 'AP — Trade'
  );

  -- Flip JE to posted (triggers balance check P0001 + closed-period check P0002).
  UPDATE public.journal_entries SET status = 'posted' WHERE id = v_je_id;

  RETURN NEW;
END;
$$;
```

```sql
  WHERE id = v_bill_id;
```

```sql
  v_old        jsonb;

BEGIN

  IF v_voided_at IS NOT NULL THEN
    RAISE EXCEPTION 'Cannot reallocate a voided payment' USING ERRCODE = 'P0004';

  DELETE FROM public.bill_payment_allocations
  WHERE payment_id = p_payment_id AND voided_at IS NULL;

  FOR v_alloc IN SELECT * FROM jsonb_array_elements(p_allocations)
  LOOP
    INSERT INTO public.bill_payment_allocations (
      payment_id, bill_id, allocated_amount
    ) VALUES (
      p_payment_id,
      (v_alloc->>'bill_id')::uuid,
      (v_alloc->>'allocated_amount')::numeric(12,2)
    );
  END LOOP;

  -- At commit, deferred constraint trigger validates P0014-P0017.

  INSERT INTO public.audit_log (actor_id, actor_type, action, resource_type, resource_id, metadata)
  VALUES (p_actor_id, 'staff', 'bill_payment.reallocated', 'bill_payment', p_payment_id,
```

```ts
type ActionResult<T> = { ok: true; data: T } | { ok: false; error: string; field?: string | null };

const BUCKET = "bill-attachments";

function sanitize(name: string): string {
  return name.replace(/[^a-zA-Z0-9._-]/g, "_").slice(0, 200);
}

export async function uploadBillAttachmentAction(
  billId: string,
  file: { name: string; mime_type: string; size_bytes: number; bytes: ArrayBuffer | Uint8Array }
): Promise<ActionResult<{ id: string; storage_path: string }>> {
  const profile = await requireAdminStaff();
  const parsed = billAttachmentUploadSchema.safeParse({
    bill_id: billId, filename: file.name, mime_type: file.mime_type, size_bytes: file.size_bytes,
  });
  if (!parsed.success) {
    const first = parsed.error.issues[0];
    return { ok: false, error: first?.message ?? "Invalid file", field: first?.path?.[0] as string };
  }

  const admin = createAdminClient();
  const path = `bills/${billId}/${crypto.randomUUID()}-${sanitize(file.name)}`;
  const { error: upErr } = await admin.storage.from(BUCKET).upload(path, file.bytes, {
    contentType: file.mime_type, upsert: false,
  });
  if (upErr) return { ok: false, error: `Upload failed: ${upErr.message}` };

  const { data, error } = await admin.from("bill_attachments").insert({
    bill_id: billId,
    storage_path: path,
    filename: sanitize(file.name),
    mime_type: file.mime_type,
    size_bytes: file.size_bytes,
    uploaded_by: profile.id,
  }).select("id").single();

  if (error) {
    await admin.storage.from(BUCKET).remove([path]);
    return { ok: false, error: "Failed to record attachment" };
  }

  await audit({
    actor_id: profile.id, actor_type: "staff", action: "bill_attachment.uploaded",
    resource_type: "bill_attachment", resource_id: data.id,
    metadata: { bill_id: billId, filename: sanitize(file.name) },
  });

  return { ok: true, data: { id: data.id, storage_path: path } };
}

export async function getBillAttachmentDownloadUrlAction(attachmentId: string): Promise<ActionResult<{ url: string }>> {
  const profile = await requireAdminStaff();
  const admin = createAdminClient();

  const { data: att } = await admin.from("bill_attachments").select("*, bills!bill_id (id)").eq("id", attachmentId).maybeSingle();
  if (!att) return { ok: false, error: "Attachment not found" };

  const { data, error } = await admin.storage.from(BUCKET).createSignedUrl(att.storage_path, 300);
  if (error || !data) return { ok: false, error: "Failed to create signed URL" };

  await audit({
    actor_id: profile.id, actor_type: "staff", action: "bill_attachment.downloaded",
    resource_type: "bill_attachment", resource_id: attachmentId, metadata: {},
  });

  return { ok: true, data: { url: data.signedUrl } };
}

export async function deleteBillAttachmentAction(attachmentId: string): Promise<ActionResult<null>> {
  const profile = await requireAdminStaff();
  const admin = createAdminClient();

  const { data: att } = await admin.from("bill_attachments")
    .select("storage_path, bills!bill_id (status)")
    .eq("id", attachmentId).maybeSingle();
  if (!att) return { ok: false, error: "Attachment not found" };

  const billStatus = (att as any).bills?.status;
  if (billStatus !== "draft") {
    return { ok: false, error: "Cannot delete attachments from a posted bill — BIR record retention applies." };
  }

  await admin.storage.from(BUCKET).remove([(att as any).storage_path]);
  await admin.from("bill_attachments").delete().eq("id", attachmentId);

  await audit({
    actor_id: profile.id, actor_type: "staff", action: "bill_attachment.deleted",
    resource_type: "bill_attachment", resource_id: attachmentId, metadata: {},
  });

  return { ok: true, data: null };
}
```

```ts
"use server";
import { requireAdminStaff } from "@/lib/auth/require-admin";
import { createAdminClient } from "@/lib/supabase/admin";
import { todayManilaISODate } from "@/lib/dates/manila";

type DashboardData = {
  outstanding_total_php: number;
  aging_buckets: { current: number; d1_30: number; d31_60: number; d60_plus: number };
  drafts: { count: number; oldest_age_days: number };
  upcoming_recurring: Array<{ id: string; vendor_name: string; description: string; next_run_date: string; amount_php: number | null }>;
  top_vendors_by_outstanding: Array<{ vendor_id: string; vendor_name: string; outstanding_php: number }>;
  wt_accumulated_this_month_php: number;
};

export async function getAPDashboardAction(): Promise<
  { ok: true; data: DashboardData } | { ok: false; error: string }
> {
  await requireAdminStaff();
  const admin = createAdminClient();
  const today = todayManilaISODate();
  const monthStart = `${today.slice(0, 7)}-01`;

  const { data: bills } = await admin.from("bills")
    .select(`id, vendor_id, due_date, outstanding_amount, wt_amount, bill_date, status, vendors!vendor_id ( name )`)
    .gt("outstanding_amount", 0)
    .neq("status", "voided");

  const bucket = (dueDate: string): keyof DashboardData["aging_buckets"] => {
    const days = Math.floor((Date.parse(today) - Date.parse(dueDate)) / 86400000);
    if (days <= 0) return "current";
    if (days <= 30) return "d1_30";
    if (days <= 60) return "d31_60";
    return "d60_plus";
  };

  const buckets = { current: 0, d1_30: 0, d31_60: 0, d60_plus: 0 };
  const byVendor = new Map<string, { name: string; outstanding: number }>();
  let outstandingTotal = 0;

  for (const b of bills ?? []) {
    const o = Number(b.outstanding_amount);
    outstandingTotal += o;
    buckets[bucket(b.due_date)] += o;
    const vn = (b.vendors as any)?.name ?? "Unknown";
    const cur = byVendor.get(b.vendor_id) ?? { name: vn, outstanding: 0 };
    cur.outstanding += o;
    byVendor.set(b.vendor_id, cur);
  }

  const topVendors = [...byVendor.entries()]
    .map(([vendor_id, v]) => ({ vendor_id, vendor_name: v.name, outstanding_php: v.outstanding }))
    .sort((a, b) => b.outstanding_php - a.outstanding_php)
    .slice(0, 5);

  const { data: drafts } = await admin.from("bills")
    .select("id, created_at").eq("status", "draft")
    .order("created_at", { ascending: true });
  const oldestDraft = drafts?.[0];
  const oldestAgeDays = oldestDraft
    ? Math.floor((Date.now() - Date.parse(oldestDraft.created_at)) / 86400000)
    : 0;

  const nextWeek = new Date(); nextWeek.setDate(nextWeek.getDate() + 7);
  const { data: upcoming } = await admin.from("recurring_bill_templates")
    .select(`id, description, next_run_date, amount_php, vendors!vendor_id ( name )`)
    .eq("is_active", true)
    .lte("next_run_date", nextWeek.toISOString().slice(0, 10))
    .order("next_run_date");
  const upcomingMapped = (upcoming ?? []).map((t: any) => ({
    id: t.id, vendor_name: t.vendors?.name ?? "Unknown",
    description: t.description, next_run_date: t.next_run_date, amount_php: t.amount_php,
  }));

  const { data: wtBills } = await admin.from("bills")
    .select("wt_amount").gte("bill_date", monthStart).neq("status", "voided");
  const wtTotal = (wtBills ?? []).reduce((s: number, b: any) => s + Number(b.wt_amount ?? 0), 0);

  return {
    ok: true,
    data: {
      outstanding_total_php: outstandingTotal,
      aging_buckets: buckets,
      drafts: { count: drafts?.length ?? 0, oldest_age_days: oldestAgeDays },
      upcoming_recurring: upcomingMapped,
      top_vendors_by_outstanding: topVendors,
      wt_accumulated_this_month_php: wtTotal,
    },
  };
}
```

```tsx
"use client";
import { useState, useEffect } from "react";
import { useRouter } from "next/navigation";
import { createBillPaymentAction } from "@/lib/actions/accounting/bill-payments";
import { listBillsAction } from "@/lib/actions/accounting/bills";
import { todayManilaISODate } from "@/lib/dates/manila";
import { formatPhp } from "@/lib/format";

type Vendor = any; type Account = any; type Bill = any;

export function PaymentFormClient({ vendors, cashAccounts }: { vendors: Vendor[]; cashAccounts: Account[] }) {
  const router = useRouter();

  const [vendorId, setVendorId] = useState("");
  const [bills, setBills] = useState<Bill[]>([]);
  const [allocations, setAllocations] = useState<Record<string, string>>({}); // bill_id -> amount string
  const [amount, setAmount] = useState("");
  const [method, setMethod] = useState<"cash"|"bank_transfer"|"gcash"|"cheque">("cash");
  const [cashAccountId, setCashAccountId] = useState("");
  const [paymentDate, setPaymentDate] = useState(todayManilaISODate());
  const [reference, setReference] = useState("");
  const [chequeNumber, setChequeNumber] = useState("");
  const [chequeDate, setChequeDate] = useState(todayManilaISODate());
  const [pending, setPending] = useState(false);
  const [error, setError] = useState<string | null>(null);

  // Load unpaid bills when vendor changes.
  useEffect(() => {
    if (!vendorId) { setBills([]); setAllocations({}); return; }
    listBillsAction({ vendor_id: vendorId, limit: 100 }).then((r) => {
      if (!r.ok) return;
      setBills(r.data.filter((b: any) => ["posted", "partially_paid"].includes(b.status)));
      setAllocations({});
    });
  }, [vendorId]);

  function setAllocation(billId: string, value: string) {
    setAllocations({ ...allocations, [billId]: value });
  }

  function autoAllocate() {
    // Fill bills oldest-first up to the entered amount.
    const remaining = Number(amount || 0);
    const sorted = [...bills].sort((a, b) => a.due_date.localeCompare(b.due_date));
    const next: Record<string, string> = {};
    let rem = remaining;
    for (const b of sorted) {
      if (rem <= 0) break;
      const o = Number(b.outstanding_amount);
      const take = Math.min(rem, o);
      next[b.id] = String(take.toFixed(2));
      rem -= take;
    }
    setAllocations(next);
  }

  const allocSum = Object.values(allocations).reduce((s, v) => s + Number(v || 0), 0);
  const allocValid = allocSum > 0 && Math.abs(allocSum - Number(amount || 0)) < 0.005;

  async function submit() {
    setPending(true); setError(null);
    const payload: any = {
      vendor_id: vendorId,
      payment_date: paymentDate,
      method, cash_account_id: cashAccountId,
      amount_php: Number(amount),
      allocations: Object.entries(allocations)
        .filter(([, v]) => Number(v) > 0)
        .map(([bill_id, v]) => ({ bill_id, allocated_amount: Number(v) })),
    };
    if (method === "cheque") {
      payload.cheque_number = chequeNumber;
      payload.cheque_date = chequeDate;
    } else if (reference) {
      payload.reference = reference;
    }

    const r = await createBillPaymentAction(payload);
    setPending(false);
    if (r.ok) router.push(`/staff/admin/accounting/ap/payments/${r.data.id}`);
    else setError(r.error);
  }

  return (
    <div className="space-y-6 p-4 max-w-3xl">
      <h1 className="text-2xl font-semibold">New payment</h1>
      {error && <div className="p-3 bg-red-50 text-red-700 rounded text-sm">{error}</div>}

      <Field label="Vendor *">
        <select required value={vendorId} onChange={(e) => setVendorId(e.target.value)}
          className="border rounded px-3 py-2 w-full min-h-[44px]">
          <option value="">Select vendor</option>
          {vendors.map((v: any) => <option key={v.id} value={v.id}>{v.name}</option>)}
        </select>
      </Field>

      {vendorId && bills.length === 0 && (
        <div className="text-sm text-gray-500">No unpaid bills for this vendor.</div>
      )}

      {bills.length > 0 && (
        <section>
          <div className="flex justify-between items-center mb-2">
            <h2 className="font-semibold">Allocate to bills</h2>
            <button type="button" onClick={autoAllocate} className="text-sm text-blue-700 underline">Auto-fill from amount</button>
          </div>
          <div className="overflow-x-auto">
            <table className="w-full min-w-[640px] border-collapse">
              <thead className="bg-gray-50"><tr>
                <th className="p-2 text-left">Bill #</th>
                <th className="p-2 text-left">Due</th>
                <th className="p-2 text-right">Outstanding</th>
                <th className="p-2 text-right">Allocate</th>
              </tr></thead>
              <tbody>
                {bills.map((b) => (
                  <tr key={b.id}>
                    <td className="p-2">{b.bill_number}</td>
                    <td className="p-2">{b.due_date}</td>
                    <td className="p-2 text-right">{formatPhp(b.outstanding_amount)}</td>
                    <td className="p-2 text-right">
                      <input type="number" step="0.01" min="0" max={b.outstanding_amount}
                        value={allocations[b.id] ?? ""} onChange={(e) => setAllocation(b.id, e.target.value)}
                        className="border rounded px-2 py-1 w-32 text-right min-h-[44px]" />
                    </td>
                  </tr>
                ))}
              </tbody>
            </table>
          </div>
          <div className={`text-right text-sm mt-2 ${allocValid ? "text-green-700" : "text-amber-700"}`}>
            Allocated: ₱{allocSum.toFixed(2)} / Payment amount: ₱{Number(amount || 0).toFixed(2)}
          </div>
        </section>
      )}

      <section className="grid grid-cols-1 md:grid-cols-2 gap-3">
        <Field label="Payment amount *">
          <input required type="number" step="0.01" min="0.01" value={amount} onChange={(e) => setAmount(e.target.value)}
            className="border rounded px-3 py-2 w-full min-h-[44px]" />
        </Field>
        <Field label="Payment date *">
          <input required type="date" value={paymentDate} max={todayManilaISODate()}
            onChange={(e) => setPaymentDate(e.target.value)}
            className="border rounded px-3 py-2 w-full min-h-[44px]" />
        </Field>
        <Field label="Method *">
          <select required value={method} onChange={(e) => setMethod(e.target.value as any)}
            className="border rounded px-3 py-2 w-full min-h-[44px]">
            {["cash","bank_transfer","gcash","cheque"].map((m) => <option key={m} value={m}>{m}</option>)}
          </select>
        </Field>
        <Field label="Cash account *">
          <select required value={cashAccountId} onChange={(e) => setCashAccountId(e.target.value)}
            className="border rounded px-3 py-2 w-full min-h-[44px]">
            <option value="">Select</option>
            {cashAccounts.map((a) => <option key={a.id} value={a.id}>{a.code} — {a.name}</option>)}
          </select>
        </Field>
        {method === "cheque" ? (
          <>
            <Field label="Cheque number *"><input required value={chequeNumber} onChange={(e) => setChequeNumber(e.target.value)} className="border rounded px-3 py-2 w-full min-h-[44px]" /></Field>
            <Field label="Cheque date *"><input required type="date" value={chequeDate} onChange={(e) => setChequeDate(e.target.value)} className="border rounded px-3 py-2 w-full min-h-[44px]" /></Field>
          </>
        ) : (
          <Field label="Reference"><input value={reference} onChange={(e) => setReference(e.target.value)} className="border rounded px-3 py-2 w-full min-h-[44px]" /></Field>
        )}
      </section>

      <div className="flex justify-end">
        <button onClick={submit} disabled={pending || !allocValid || !vendorId}
          className="px-4 py-2 bg-blue-600 text-white rounded min-h-[44px] disabled:opacity-50">
          {pending ? "Saving..." : "Record payment"}
        </button>
      </div>
    </div>
  );
}

function Field({ label, children }: { label: string; children: React.ReactNode }) {
  return <label className="block"><span className="text-sm text-gray-700 mb-1 block">{label}</span>{children}</label>;
}
```

```ts
import { createAdminClient } from "@/lib/supabase/admin";
import { reportError } from "@/lib/observability/report-error";
import { todayManilaISODate } from "@/lib/dates/manila";
import { audit } from "@/lib/audit/log";

export const dynamic = "force-dynamic";

const MAX_ITERATIONS_PER_TEMPLATE = 12;

// Vercel Cron sends GET by default. If you change vercel.json to use POST,
// rename this export to `POST` (App Router routes the HTTP method by export name).
export async function GET(request: Request) {
  // 1. Validate cron secret.
  const auth = request.headers.get("authorization") ?? "";
  if (auth !== `Bearer ${process.env.CRON_SECRET}`) {
    return Response.json({ error: "unauthorized" }, { status: 401 });
  }

  const today = todayManilaISODate();
  const admin = createAdminClient();

  // 2. Query due templates.
  const { data: templates, error } = await admin.from("recurring_bill_templates")
    .select("id")
    .eq("is_active", true)
    .lte("next_run_date", today);

  if (error) {
    reportError(error, { tag: "cron/recurring-bills:query" });
    return Response.json({ error: "query failed" }, { status: 500 });
  }

  let processed = 0;
  let draftsCreated = 0;
  const failures: Array<{ template_id: string; error: string }> = [];

  // 3. Per-template loop with isolation.
  for (const t of templates ?? []) {
    processed += 1;
    try {
      let iters = 0;
      while (iters < MAX_ITERATIONS_PER_TEMPLATE) {
        const { data, error: rpcErr } = await admin.rpc("ap_post_recurring_template", { p_template_id: t.id });
        if (rpcErr) throw new Error(rpcErr.message);
        if ((data as any)?.skipped) break;
        draftsCreated += 1;
        iters += 1;
      }
      if (iters === MAX_ITERATIONS_PER_TEMPLATE) {
        throw new Error(`Iteration limit reached (${MAX_ITERATIONS_PER_TEMPLATE}); next_run_date may be stuck`);
      }
    } catch (err: any) {
      reportError(err, { tag: "cron/recurring-bills:template", template_id: t.id });
      await audit({
        actor_id: null, actor_type: "system", action: "recurring_template.fire_failed",
        resource_type: "recurring_bill_template", resource_id: t.id,
        metadata: { error: err.message, run_date: today },
      });
      failures.push({ template_id: t.id, error: err.message });
    }
  }

  return Response.json({ processed, drafts_created: draftsCreated, failures });
}
```

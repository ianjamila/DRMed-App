# 12.5 COGS + Doctor PF Subledger Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship the COGS + Doctor PF subledger described in spec `docs/superpowers/specs/2026-05-26-12.5-cogs-doctor-pf-design.md` — line-level PF accrual with hybrid cash/HMO recognition via a new `2160` holding account, end-of-day batch payouts, send-out per-test COGS accrual with Hi Precision true-up matching, plus surgical rewrite of the existing op→GL release bridge.

```sql
-- physicians.compensation_arrangement drives the visit form's clinic_fee_php
```

```sql
  test_request_id       uuid not null references public.test_requests(id),
  physician_id          uuid not null references public.physicians(id),
  pf_php                numeric(10,2) not null check (pf_php != 0),
  recognition_basis     text not null check (recognition_basis in
    ('cash_at_release', 'hmo_at_settlement', 'clawback')),
  recognized_at         timestamptz,

-- Hot path: EOD payouts grouped by physician.

  on public.doctor_pf_entries(physician_id, recognized_at)
```

- [ ] **Step 1: Append the batch-number counter function**

```sql

-- =============================================================================
-- Section 5 — Helpers.
-- =============================================================================

-- Increment-and-return for PF disbursement batch numbers. Called by
-- trg_bridge_pf_disbursement_post (Section 6) before drafting the JE.
-- Returns the assigned batch_number; updates the counter row atomically.
create or replace function public.next_pf_disbursement_batch_number(p_year smallint)
returns bigint
language plpgsql
security definer
set search_path = public
as $$
declare
  v_n bigint;
begin
  insert into public.pf_disbursement_year_counters(year, next_n)
    values (p_year, 1)
    on conflict (year) do update set next_n = pf_disbursement_year_counters.next_n + 1
    returning next_n into v_n;
  -- The ON CONFLICT branch returns the incremented next_n; the INSERT branch
  -- returns 1. Both are correct: first row of year = 1, second = 2, etc.
  -- Actually wait — INSERT branch returns the inserted 1, then NEXT call hits
  -- conflict and returns 2. That's right.
  return v_n;
end;
$$;

revoke all on function public.next_pf_disbursement_batch_number(smallint) from public;
```

```sql
-- bridge_test_request_released — split the consult/procedure release JE into
-- clinic_fee revenue + doctor_pf liability (or pending-HMO liability), plus
-- send-out COGS accrual when applicable. See spec §4.1.
create or replace function public.bridge_test_request_released()
returns trigger
language plpgsql
security definer
set search_path = public
as $$
declare
  v_tr               record;
  v_visit            record;
  v_service          record;
  v_physician_id     uuid;
  v_actor            uuid := auth.uid();
  v_je_id            uuid;
  v_je_number        text;
  v_year             smallint := extract(year from new.released_at)::smallint;
  v_cash_account     text;
  v_revenue_account  text;
  v_discount_account text;
  v_posting_date     date := new.released_at::date;
  v_line_order       int := 1;
begin
  -- Only fire on status→released transitions.
  if not (old.status is distinct from new.status and new.status = 'released') then
    return new;
  end if;

  -- Idempotency: if a JE already exists for this source, do nothing.
  if exists (
    select 1 from public.journal_entries
    where source_kind = 'test_request' and source_id = new.id and status = 'posted'
  ) then
    return new;
  end if;

  v_tr := new;
  select * into v_visit from public.visits where id = v_tr.visit_id;
  select * into v_service from public.services where id = v_tr.service_id;

  -- Guard P0034 — attending physician required for consult/procedure.
  if v_service.kind in ('doctor_consultation', 'doctor_procedure') then
    v_physician_id := coalesce(v_tr.attending_physician_id, v_visit.attending_physician_id);
    if v_physician_id is null then
      raise exception 'attending_physician_id required for consult/procedure release on test_request %', v_tr.id
        using errcode = 'P0034';
    end if;
  end if;

  -- Resolve revenue + discount accounts (per existing 0031 mapper).
  v_revenue_account := case v_service.kind
    when 'lab_test'              then '4100'
    when 'lab_package'           then '4100'
    when 'doctor_consultation'   then '4200'
    when 'doctor_procedure'      then '4500'
    when 'vaccine'               then '4100'
    when 'home_service'          then '4100'
    when 'send_out'              then '4100'  -- legacy enum value
    when 'diagnostic_package'    then '4100'
    else null
  end;

  if v_revenue_account is null then
    -- Unknown kind — book to Suspense + audit (matches 0033 pattern).
    v_revenue_account := '9999';
    insert into public.audit_log (actor_id, actor_type, action, resource_type, resource_id, metadata)
    values (v_actor, 'system', 'release.unknown_service_kind',
            'test_request', v_tr.id,
            jsonb_build_object('service_kind', v_service.kind));
  end if;

  v_discount_account := case v_service.kind
    when 'doctor_consultation' then '4920'
    when 'doctor_procedure'    then '4920'
    else '4910'
  end;

  -- Resolve cash/AR account.
  if v_visit.hmo_provider_id is not null then
    v_cash_account := '1110';  -- AR HMO
  else
    v_cash_account := '1100';  -- AR Patients; payment-side JE moves to cash on receipt
  end if;

  -- Draft the JE header.
  v_je_number := public.next_je_number(v_year);
  insert into public.journal_entries(
    entry_number, posting_date, status, source_kind, source_id,
    description, created_by
  ) values (
    v_je_number, v_posting_date, 'draft', 'test_request', v_tr.id,
    'Release of test_request ' || v_tr.id, v_actor
  ) returning id into v_je_id;

  -- Line: DR cash/AR for final_price_php
  if v_tr.final_price_php > 0 then
    insert into public.journal_lines(entry_id, line_order, account_id, debit_php, credit_php, description)
    values (v_je_id, v_line_order, public.coa_uuid_for_code(v_cash_account),
            v_tr.final_price_php, 0,
            'Release receivable');
    v_line_order := v_line_order + 1;
  end if;

  -- Consult/procedure split: clinic_fee → revenue, doctor_pf → 2110 (cash) or 2160 (HMO).
  if v_service.kind in ('doctor_consultation', 'doctor_procedure') then
    -- CR revenue for clinic_fee (skip 0 lines).
    if coalesce(v_tr.clinic_fee_php, 0) > 0 then
      insert into public.journal_lines(entry_id, line_order, account_id, debit_php, credit_php, description)
      values (v_je_id, v_line_order, public.coa_uuid_for_code(v_revenue_account),
              0, v_tr.clinic_fee_php, 'Clinic fee');
      v_line_order := v_line_order + 1;
    end if;

    -- CR PF to 2110 or 2160 (skip 0 lines).
    if coalesce(v_tr.doctor_pf_php, 0) > 0 then
      if v_visit.hmo_provider_id is null then
        -- Cash path: accrue directly to 2110 AP — Doctors.
        insert into public.journal_lines(entry_id, line_order, account_id, debit_php, credit_php, description)
        values (v_je_id, v_line_order, public.coa_uuid_for_code('2110'),
                0, v_tr.doctor_pf_php, 'Doctor PF accrual (cash)');
        v_line_order := v_line_order + 1;

        insert into public.doctor_pf_entries(
          test_request_id, physician_id, pf_php, recognition_basis,
          recognized_at, journal_entry_id
        ) values (
          v_tr.id, v_physician_id, v_tr.doctor_pf_php, 'cash_at_release',
          now(), v_je_id
        );
      else
        -- HMO path: park in 2160 Pending HMO Settlement.
        insert into public.journal_lines(entry_id, line_order, account_id, debit_php, credit_php, description)
        values (v_je_id, v_line_order, public.coa_uuid_for_code('2160'),
                0, v_tr.doctor_pf_php, 'Doctor PF pending HMO');
        v_line_order := v_line_order + 1;

        insert into public.doctor_pf_entries(
          test_request_id, physician_id, pf_php, recognition_basis,
          recognized_at, journal_entry_id
        ) values (
          v_tr.id, v_physician_id, v_tr.doctor_pf_php, 'hmo_at_settlement',
          null, null  -- recognition + JE fire at settlement
        );
      end if;
    end if;
  else
    -- Non-consult/procedure: CR revenue for the full final_price.
    if v_tr.final_price_php > 0 then
      insert into public.journal_lines(entry_id, line_order, account_id, debit_php, credit_php, description)
      values (v_je_id, v_line_order, public.coa_uuid_for_code(v_revenue_account),
              0, v_tr.final_price_php, 'Release revenue');
      v_line_order := v_line_order + 1;
    end if;
  end if;

  -- Discount line (DR contra-revenue) — unchanged from 0030 behavior.
  if coalesce(v_tr.discount_amount_php, 0) > 0 then
    insert into public.journal_lines(entry_id, line_order, account_id, debit_php, credit_php, description)
    values (v_je_id, v_line_order, public.coa_uuid_for_code(v_discount_account),
            v_tr.discount_amount_php, 0, 'Discount');
    v_line_order := v_line_order + 1;
  end if;

  -- Send-out COGS accrual: add lines if service is_send_out AND unit_cost is set.
  if v_service.is_send_out then
    if v_service.send_out_unit_cost_php is not null and v_service.send_out_unit_cost_php > 0 then
      insert into public.journal_lines(entry_id, line_order, account_id, debit_php, credit_php, description)
      values (v_je_id, v_line_order, public.coa_uuid_for_code('6420'),
              v_service.send_out_unit_cost_php, 0, 'Send-out COGS');
      v_line_order := v_line_order + 1;
      insert into public.journal_lines(entry_id, line_order, account_id, debit_php, credit_php, description)
      values (v_je_id, v_line_order, public.coa_uuid_for_code('2150'),
              0, v_service.send_out_unit_cost_php, 'Accrued send-out');
      v_line_order := v_line_order + 1;

      insert into public.cogs_send_out_entries(
        test_request_id, service_id, vendor_id, unit_cost_php, journal_entry_id
      ) values (
        v_tr.id, v_service.id, v_service.send_out_vendor_id,
        v_service.send_out_unit_cost_php, v_je_id
      );
    else
      -- D10: do NOT block release. Record entry with unit_cost=0 + audit.
      insert into public.cogs_send_out_entries(
        test_request_id, service_id, vendor_id, unit_cost_php, journal_entry_id
      ) values (
        v_tr.id, v_service.id, null, 0, null
      );
      insert into public.audit_log(actor_id, actor_type, action, resource_type, resource_id, metadata)
      values (v_actor, 'system', 'send_out.unit_cost_missing',
              'test_request', v_tr.id,
              jsonb_build_object('service_id', v_service.id, 'service_code', v_service.code));
    end if;
  end if;

  -- Post the JE.
  update public.journal_entries set status = 'posted' where id = v_je_id;

  return new;
end;
$$;

revoke all on function public.bridge_test_request_released() from public;
```

```sql
-- Section 6.2 — bridge_test_request_cancelled — preserve existing reversal-JE
-- logic AND soft-void any subledger rows attached to this test_request.
-- The existing 0030 function already mirrors lines for the reversal JE; this
-- adds the subledger cleanup. We CREATE OR REPLACE the existing function with
-- the additional UPDATEs.
create or replace function public.bridge_test_request_cancelled()
returns trigger
language plpgsql
security definer
set search_path = public
as $$
declare
  v_actor       uuid := auth.uid();
  v_orig_je     record;
  v_rev_je_id   uuid;
  v_je_number   text;
  v_year        smallint := extract(year from coalesce(new.cancelled_at, now()))::smallint;
  v_line        record;
  v_line_order  int := 1;
begin
  if not (old.status is distinct from new.status and new.status = 'cancelled') then
    return new;
  end if;

  -- Find the original posted JE (release-side).
  select * into v_orig_je from public.journal_entries
  where source_kind = 'test_request' and source_id = new.id and status = 'posted'
  limit 1;

  if v_orig_je.id is null then
    return new;  -- never released; nothing to reverse
  end if;

  -- Draft reversal.
  v_je_number := public.next_je_number(v_year);
  insert into public.journal_entries(
    entry_number, posting_date, status, source_kind, source_id,
    description, created_by, reverses
  ) values (
    v_je_number, coalesce(new.cancelled_at::date, current_date),
    'draft', 'reversal', null,
    'Reversal of ' || v_orig_je.entry_number, v_actor, v_orig_je.id
  ) returning id into v_rev_je_id;

  -- Mirror lines.
  for v_line in
    select * from public.journal_lines where entry_id = v_orig_je.id order by line_order
  loop
    insert into public.journal_lines(entry_id, line_order, account_id, debit_php, credit_php, description)
    values (
      v_rev_je_id, v_line_order, v_line.account_id,
      v_line.credit_php, v_line.debit_php,
      'Reversal: ' || coalesce(v_line.description, '')
    );
    v_line_order := v_line_order + 1;
  end loop;

  update public.journal_entries set status = 'posted' where id = v_rev_je_id;
  update public.journal_entries set status = 'reversed', reversed_by = v_rev_je_id
    where id = v_orig_je.id;

  -- 12.5 addition: soft-void subledger rows.
  update public.doctor_pf_entries
    set voided_at = now(),
        voided_by = v_actor,
        void_reason = 'test_request_cancelled'
    where test_request_id = new.id and voided_at is null;

  update public.cogs_send_out_entries
    set voided_at = now(),
        voided_by = v_actor,
        void_reason = 'test_request_cancelled'
    where test_request_id = new.id and voided_at is null;

  return new;
end;
$$;
```

```sql
-- Section 6.3 — Fire PF settlement JE when HMO actually pays. Trigger fires
-- on hmo_payment_allocations INSERT (NOT hmo_claim_resolutions — see spec §4.3).
create or replace function public.bridge_pf_at_hmo_allocation()
returns trigger
language plpgsql
security definer
set search_path = public
as $$
declare
  v_actor          uuid := auth.uid();
  v_item           record;
  v_tr             record;
  v_pfe            record;
  v_payment        record;
  v_settlement_ratio numeric(10,6);
  v_pf_to_accrue   numeric(10,2);
  v_je_id          uuid;
  v_je_number      text;
  v_year           smallint;
  v_posting_date   date;
begin
  -- Look up the item and its test_request.
  select * into v_item from public.hmo_claim_items where id = new.item_id;
  if v_item.test_request_id is null then return new; end if;

  select * into v_tr from public.test_requests where id = v_item.test_request_id;

  -- Find a pending HMO PF entry for this test_request.
  select * into v_pfe from public.doctor_pf_entries
  where test_request_id = v_tr.id
    and recognition_basis = 'hmo_at_settlement'
    and recognized_at is null
    and voided_at is null
  limit 1;

  if v_pfe.id is null then return new; end if;  -- no pending PF; nothing to do

  -- Compute proportional accrual. In practice the ratio is 1.0 or 0.0 (per
  -- spec D7); the formula handles partials gracefully.
  if v_item.billed_amount_php = 0 then return new; end if;
  v_settlement_ratio := new.amount_php / v_item.billed_amount_php;
  v_pf_to_accrue := round(coalesce(v_tr.doctor_pf_php, 0) * v_settlement_ratio, 2);

  if v_pf_to_accrue <= 0 then return new; end if;

  -- Derive posting_date from parent payment.
  select * into v_payment from public.payments where id = new.payment_id;
  v_posting_date := coalesce(v_payment.posting_date, current_date);
  v_year := extract(year from v_posting_date)::smallint;

  -- Draft + post JE: DR 2160 / CR 2110.
  v_je_number := public.next_je_number(v_year);
  insert into public.journal_entries(
    entry_number, posting_date, status, source_kind, source_id,
    description, created_by
  ) values (
    v_je_number, v_posting_date, 'draft', 'doctor_pf_accrual', v_pfe.id,
    'HMO PF settlement: 2160 → 2110', v_actor
  ) returning id into v_je_id;

  insert into public.journal_lines(entry_id, line_order, account_id, debit_php, credit_php, description)
  values (v_je_id, 1, public.coa_uuid_for_code('2160'),
          v_pf_to_accrue, 0, 'Reclass from PF pending');
  insert into public.journal_lines(entry_id, line_order, account_id, debit_php, credit_php, description)
  values (v_je_id, 2, public.coa_uuid_for_code('2110'),
          0, v_pf_to_accrue, 'Doctor PF accrued at HMO settlement');

  update public.journal_entries set status = 'posted' where id = v_je_id;

  -- Update the PF entry with settlement details + snapshot the actual accrued amount.
  update public.doctor_pf_entries
    set recognized_at = now(),
        journal_entry_id = v_je_id,
        hmo_allocation_id = new.id,
        pf_php = v_pf_to_accrue
    where id = v_pfe.id;

  return new;
end;
$$;
```

```sql
-- Section 6.5 — When a cash payment is voided AND the visit becomes un-paid,
-- reverse PF accruals on cash entries OR insert clawback rows if PF was
-- already disbursed (Risk #2 (b)). See spec §4.5.
create or replace function public.bridge_payment_void_pf_cascade()
returns trigger
language plpgsql
security definer
set search_path = public
as $$
declare
  v_actor       uuid := auth.uid();
  v_visit       record;
  v_tr          record;
  v_pfe         record;
  v_je_id       uuid;
  v_je_number   text;
  v_year        smallint;
begin
  -- Only act on payments going from active to voided.
  if not (old.voided_at is null and new.voided_at is not null) then
    return new;
  end if;

  -- If visit is still paid via other payments, do nothing.
  select * into v_visit from public.visits where id = new.visit_id;
  if v_visit.payment_status = 'paid' then return new; end if;

  -- Walk all test_requests on this visit; for each cash-recognized PF entry,
  -- either reverse-JE-and-soft-void OR insert a clawback row.
  for v_tr in (select * from public.test_requests where visit_id = new.visit_id) loop
    for v_pfe in (
      select * from public.doctor_pf_entries
      where test_request_id = v_tr.id
        and recognition_basis = 'cash_at_release'
        and voided_at is null
    ) loop
      if v_pfe.disbursement_id is not null then
        -- PF already paid out. Insert clawback row + audit; surface in UI.
        insert into public.doctor_pf_entries(
          test_request_id, physician_id, pf_php, recognition_basis,
          recognized_at, journal_entry_id
        ) values (
          v_pfe.test_request_id, v_pfe.physician_id, -v_pfe.pf_php,
          'clawback', now(), null
        );
        insert into public.audit_log(actor_id, actor_type, action, resource_type, resource_id, metadata)
        values (v_actor, 'system', 'pf_clawback.alert',
                'doctor_pf_entries', v_pfe.id,
                jsonb_build_object('visit_id', new.visit_id,
                                   'physician_id', v_pfe.physician_id,
                                   'pf_php', v_pfe.pf_php));
      else
        -- PF not yet disbursed. Draft a partial reversal JE (just the PF lines).
        v_year := extract(year from now())::smallint;
        v_je_number := public.next_je_number(v_year);
        insert into public.journal_entries(
          entry_number, posting_date, status, source_kind, source_id,
          description, created_by, reverses
        ) values (
          v_je_number, current_date, 'draft', 'reversal', null,
          'PF partial reversal due to payment void', v_actor, v_pfe.journal_entry_id
        ) returning id into v_je_id;

        -- Reverse just the 2110 credit (we keep the 4200 revenue as recognized — the
        -- clinic_fee is unrelated to payment fraud; only the PF disbursement piece moves).
        -- Cash side moves via the existing bridge_payment_void function already.
        insert into public.journal_lines(entry_id, line_order, account_id, debit_php, credit_php, description)
        values (v_je_id, 1, public.coa_uuid_for_code('2110'),
                v_pfe.pf_php, 0, 'Reverse PF (payment voided)');
        insert into public.journal_lines(entry_id, line_order, account_id, debit_php, credit_php, description)
        values (v_je_id, 2, public.coa_uuid_for_code('1100'),
                0, v_pfe.pf_php, 'Reverse AR (PF portion)');

        update public.journal_entries set status = 'posted' where id = v_je_id;

        update public.doctor_pf_entries
          set voided_at = now(),
              voided_by = v_actor,
              void_reason = 'payment_voided'
          where id = v_pfe.id;
      end if;
    end loop;
  end loop;

  return new;
end;
$$;
```

```sql
-- Section 6.7 — On INSERT into cogs_send_out_trueups, emit variance JE
-- (4 cases per spec §4.7). When billed=accrued, no JE — just mark entries.
create or replace function public.bridge_cogs_send_out_trueup()
returns trigger
language plpgsql
security definer
set search_path = public
as $$
declare
  v_actor       uuid := auth.uid();
  v_variance    numeric(12,2);
  v_je_id       uuid;
  v_je_number   text;
  v_year        smallint := extract(year from new.matched_at)::smallint;
begin
  -- Mark matching entries trued-up regardless of variance.
  update public.cogs_send_out_entries
    set trueup_id = new.id, trued_up_at = now()
    where vendor_id = new.vendor_id
      and accrued_at::date between new.period_start_date and new.period_end_date
      and trueup_id is null
      and voided_at is null;

  v_variance := new.variance_php;

  if v_variance = 0 then return new; end if;

  v_je_number := public.next_je_number(v_year);
  insert into public.journal_entries(
    entry_number, posting_date, status, source_kind, source_id,
    description, created_by
  ) values (
    v_je_number, new.matched_at::date, 'draft', 'cogs_send_out_trueup', new.id,
    'Send-out variance true-up (vendor ' || new.vendor_id || ')', v_actor
  ) returning id into v_je_id;

  if v_variance > 0 then
    -- Under-accrued: book more COGS.
    insert into public.journal_lines(entry_id, line_order, account_id, debit_php, credit_php, description)
    values (v_je_id, 1, public.coa_uuid_for_code('6420'), v_variance, 0, 'COGS under-accrual catchup');
    insert into public.journal_lines(entry_id, line_order, account_id, debit_php, credit_php, description)
    values (v_je_id, 2, public.coa_uuid_for_code('2150'), 0, v_variance, 'Top up accrued send-out');
  else
    -- Over-accrued: reverse some COGS.
    insert into public.journal_lines(entry_id, line_order, account_id, debit_php, credit_php, description)
    values (v_je_id, 1, public.coa_uuid_for_code('2150'), abs(v_variance), 0, 'Reverse over-accrual');
    insert into public.journal_lines(entry_id, line_order, account_id, debit_php, credit_php, description)
    values (v_je_id, 2, public.coa_uuid_for_code('6420'), 0, abs(v_variance), 'Reverse COGS');
  end if;

  update public.journal_entries set status = 'posted' where id = v_je_id;
  update public.cogs_send_out_trueups set journal_entry_id = v_je_id where id = new.id;

  return new;
end;
$$;
```

### Task 25: Create `src/lib/actions/accounting/pf-disbursements.ts`

**Files:**

- [ ] **Step 1: Write `createPfDisbursement` action**

```typescript
"use server";

import { z } from "zod";
import { createAdminClient } from "@/lib/supabase/admin";
import { requireAdminStaff } from "@/lib/auth/require-admin";
import { audit } from "@/lib/audit/log";
import { translatePgError } from "@/lib/accounting/pg-errors";
import { PfDisbursementCreateSchema } from "@/lib/validations/accounting";

type ActionResult<T = unknown> =
  | { ok: true; data: T }
  | { ok: false; error: string };

export async function createPfDisbursement(
  input: z.infer<typeof PfDisbursementCreateSchema>
): Promise<ActionResult<{ disbursement_id: string; batch_number: number }>> {
  const staff = await requireAdminStaff();
  const parsed = PfDisbursementCreateSchema.safeParse(input);
  if (!parsed.success) {
    return { ok: false, error: parsed.error.issues[0]?.message ?? "Invalid input" };
  }
  const data = parsed.data;

  const admin = createAdminClient();

  // Assign batch_number via the counter function.
  const year = new Date(data.posted_date).getFullYear();
  const { data: nRow, error: nErr } = await admin.rpc(
    "next_pf_disbursement_batch_number",
    { p_year: year }
  );
  if (nErr) return { ok: false, error: translatePgError(nErr) };
  const batchNumber = nRow as number;

  // Mark selected entries as belonging to this disbursement (we set disbursement_id
  // BEFORE inserting the header so the FK is consistent at trigger time).
  // Compute server-side total for safety; client-side total is a hint only.
  const { data: entries, error: entErr } = await admin
    .from("doctor_pf_entries")
    .select("id, pf_php, physician_id, disbursement_id, voided_at, recognized_at")
    .in("id", data.entry_ids);
  if (entErr) return { ok: false, error: translatePgError(entErr) };
  if (!entries || entries.length !== data.entry_ids.length) {
    return { ok: false, error: "One or more PF entries not found" };
  }
  for (const e of entries) {
    if (e.physician_id !== data.physician_id) {
      return { ok: false, error: "PF entries must all belong to the same physician" };
    }
    if (e.disbursement_id || e.voided_at || !e.recognized_at) {
      return { ok: false, error: "One or more PF entries are not open for disbursement" };
    }
  }
  const computedTotal = entries.reduce((s, e) => s + Number(e.pf_php), 0);
  if (Math.abs(computedTotal - data.total_php) > 0.005) {
    return { ok: false, error: `Total mismatch: expected ${computedTotal}, got ${data.total_php}` };
  }

  // Insert disbursement header — trigger emits JE.
  const { data: disb, error: insErr } = await admin
    .from("doctor_pf_disbursements")
    .insert({
      batch_number: batchNumber,
      physician_id: data.physician_id,
      posted_date: data.posted_date,
      method: data.method,
      total_php: data.total_php,
      recorded_by: staff.id,
      notes: data.notes ?? null,
    })
    .select("id, batch_number")
    .single();
  if (insErr || !disb) return { ok: false, error: translatePgError(insErr) };

  // Link entries to the disbursement.
  const { error: updErr } = await admin
    .from("doctor_pf_entries")
```

```typescript
export async function voidPfDisbursement(input: {
  disbursement_id: string;
  void_reason: string;
}): Promise<ActionResult<{ voided: true }>> {
  const staff = await requireAdminStaff();
  if (!input.disbursement_id || !input.void_reason || input.void_reason.length < 3) {
    return { ok: false, error: "Disbursement id and void reason required" };
  }

  const admin = createAdminClient();

  // Fetch disbursement + entries.
  const { data: disb, error: dErr } = await admin
    .from("doctor_pf_disbursements")
    .select("id, journal_entry_id, voided_at, total_php, physician_id")
    .eq("id", input.disbursement_id)
    .single();
  if (dErr || !disb) return { ok: false, error: translatePgError(dErr) };
  if (disb.voided_at) return { ok: false, error: "Disbursement already voided" };

  // Draft-flip pattern for JE reversal (see feedback_je_cleanup_pattern.md):
  //   1. Update original JE to status='draft'
  //   2. Insert reversal JE
  //   3. Update original back to status='reversed'
  // Done inside a transaction via supabase.rpc — would normally be a PG function,
  // but for simplicity here we use multiple admin queries. If contention is a
  // concern, extract to a SECURITY DEFINER PG function later.

  if (disb.journal_entry_id) {
    // Step 1: original to draft
    await admin
      .from("journal_entries")
      .update({ status: "draft" })
      .eq("id", disb.journal_entry_id);

    // Fetch lines of original
    const { data: lines } = await admin
      .from("journal_lines")
      .select("account_id, debit_php, credit_php, description, line_order")
      .eq("entry_id", disb.journal_entry_id)
      .order("line_order");

    // Insert reversal JE
    const year = new Date().getFullYear();
    const { data: nRow } = await admin.rpc("next_je_number", { p_year: year });
    const revEntryNumber = nRow as string;
    const { data: revJe } = await admin
      .from("journal_entries")
      .insert({
        entry_number: revEntryNumber,
        posting_date: new Date().toISOString().slice(0, 10),
        status: "draft",
        source_kind: "reversal",
        source_id: null,
        description: `Void of PF disbursement (batch PF-${disb.id})`,
        created_by: staff.id,
        reverses: disb.journal_entry_id,
      })
      .select("id")
      .single();

    if (revJe && lines) {
      for (const l of lines) {
        await admin.from("journal_lines").insert({
          entry_id: revJe.id,
          line_order: l.line_order,
          account_id: l.account_id,
          debit_php: l.credit_php,
          credit_php: l.debit_php,
          description: `Reversal: ${l.description ?? ""}`,
        });
      }
      await admin.from("journal_entries").update({ status: "posted" }).eq("id", revJe.id);
    }

    await admin
      .from("journal_entries")
      .update({ status: "reversed", reversed_by: revJe?.id ?? null })
      .eq("id", disb.journal_entry_id);
  }

  // Soft-void the disbursement + unlink entries.
  await admin
    .from("doctor_pf_disbursements")
    .update({
      voided_at: new Date().toISOString(),
      voided_by: staff.id,
      void_reason: input.void_reason,
    })
    .eq("id", input.disbursement_id);

  await admin
    .from("doctor_pf_entries")
    .update({ disbursement_id: null })
    .eq("disbursement_id", input.disbursement_id);

  await audit({
    actor_id: staff.id,
    actor_type: "staff",
    action: "pf_disbursement.voided",
    resource_type: "doctor_pf_disbursements",
    resource_id: input.disbursement_id,
    metadata: { void_reason: input.void_reason, total_php: disb.total_php },
  });

  return { ok: true, data: { voided: true } };
}
```

```typescript
  const staff = await requireAdminStaff();
  const parsed = PfBulkPayoutSchema.safeParse(input);
  if (!parsed.success) {
    return { ok: false, error: parsed.error.issues[0]?.message ?? "Invalid input" };

  const data = parsed.data;
  const admin = createAdminClient();

  const created: string[] = [];

  // Atomic: if any single physician fails, void the previously-created disbursements.
  // (For simplicity here, we iterate and fail-fast; production-grade transactional
  // semantics would push this into a SECURITY DEFINER PG function. Acceptable for v1.)

        await admin
          .from("doctor_pf_disbursements")
          .update({ voided_at: new Date().toISOString(), voided_by: staff.id, void_reason: "bulk_failed" })
          .eq("id", id);
      }
      return { ok: false, error: translatePgError(nErr) };
    }
    const { data: disb, error: insErr } = await admin
      .from("doctor_pf_disbursements")
      .insert({
        batch_number: nRow as number,
        physician_id: phys.physician_id,
        posted_date: data.posted_date,
        method: "cash",
        total_php: phys.total_php,
        recorded_by: staff.id,
        notes: `Bulk EOD payout ${data.posted_date}`,
      })
      .select("id")
      .single();
    if (insErr || !disb) {
      for (const id of created) {
        await admin
          .from("doctor_pf_disbursements")
          .update({ voided_at: new Date().toISOString(), voided_by: staff.id, void_reason: "bulk_failed" })
          .eq("id", id);
      }
      return { ok: false, error: translatePgError(insErr) };
    }
    created.push(disb.id);
    await admin
      .from("doctor_pf_entries")
      .update({ disbursement_id: disb.id })
      .in("id", phys.entry_ids);
  }

  await audit({
    actor_id: staff.id,
    actor_type: "staff",
    action: "pf_disbursement.created",
    resource_type: "doctor_pf_disbursements",
    resource_id: null,
    metadata: { bulk: true, count: created.length, posted_date: data.posted_date },
```

```sql
-- Admin-triggered scrub for clinic_fee_php on unreleased test_requests where

create or replace function public.recompute_clinic_fee_for_unreleased()
returns jsonb
language plpgsql
security definer
set search_path = public
as $$
declare
  v_affected int;
begin
  with updated as (
    update public.test_requests tr
      set clinic_fee_php = 0,
          doctor_pf_php = tr.final_price_php
      from public.visits v
```

```tsx
function ageBucket(d: string): "0_30" | "31_60" | "61_90" | "90_plus" {
  const days = (Date.now() - new Date(d).getTime()) / 86400000;
  if (days < 31) return "0_30";
  if (days < 61) return "31_60";
  if (days < 91) return "61_90";
  return "90_plus";
}
```
